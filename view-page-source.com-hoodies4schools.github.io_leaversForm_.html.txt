function openTab(tabId) {
    var tabs = document.getElementsByClassName("tab-content");
    for (var i = 0; i < tabs.length; i++) {
        tabs[i].style.opacity = 0; // Set opacity to 0 to fade out the tab content
    };

    setTimeout(() => {
        for (var i = 0; i < tabs.length; i++) {
            tabs[i].style.display = "none"; // Hide all tab content
        }
        document.getElementById(tabId).style.display = "block"; // Show the selected tab content
        setTimeout(() => {
            document.getElementById(tabId).style.opacity = 1; // Set opacity to 1 to fade in the selected tab content
        }, 50); // Shorter delay before fading in
    }, 0); // Shorter delay before changing the display
}
